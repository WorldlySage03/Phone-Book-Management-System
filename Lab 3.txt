1 - Write a program to implement the push and pop operation of a stack
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define el "\n"
#define sp " "

const int maxn = 1000000;
int a[maxn], n = 0;

int menu()
{
      int choice;

      cout << "Menu:" << el;
      cout << "   -> To push input 1." << el;
      cout << "   -> To pop input 2." << el << el;
      cout << "   -> To exit input 0." << el << el;
      cout << "Enter your choice: ";

      cin >> choice;
      return choice;
}

void push()
{
      int t;
      cout << "Input the element you want to push: ";
      cin >> t;
      cout << el;

      if (n >= maxn)
      {
            cout << "Overflow." << el;
            return;
      }

      a[n] = t;
      n++;
}

void pop()
{
      if (n == 0)
      {
            cout << "Underflow." << el;
            return;
      }

      cout << "Popping the element: " << a[n - 1] << el;
      n--;
}

void display()
{
      if (n == 0)
      {
            cout << "The stack is empty." << el << el;
            return;
      }
      else if (n > maxn)
      {
            cout << "The stack is full." << el << el;
            return;
      }

      cout << "The number of elements in the stack is: " << n << el;
      cout << "The elements from bottom to top are: ";

      for (int i = 0; i < n; i++)
      {
            cout << a[i] << sp;
      }
      cout << el << el;
}

int main()
{
      int choice;

      do
      {
            choice = menu();

            switch (choice)
            {
            case 0:
                  cout << "End of operations." << el;
                  break;
            case 1:
                  push();
                  display();
                  break;
            case 2:
                  pop();
                  display();
                  break;
            default:
                  cout << "Invalid input." << el << el;
            }
      }
      while (choice != 0);
}

2 - Write a program to evaluate a Postfix expression.
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define el "\n"
#define sp " "

const int maxn = 1000000;
int a[maxn];

int main()
{
      string expression;
      getline(cin, expression);

      for (int i = 0, j = 0; i < expression.length(); i++)
      {
            if (expression[i] == ' ') continue;
            else if (isdigit(expression[i]))
            {
                  int num = 0;
                  while (i < expression.length() && isdigit(expression[i])) 
                  {
                        num = num * 10 + (expression[i] - '0');
                        i++;
                  }
                  
                  a[j] = num;
                  j++;
                  i--;
            }
            else
            {
                  if (expression[i] == '+') a[j - 2] = a[j - 2] + a[j - 1];
                  else if (expression[i] == '-') a[j - 2] = a[j - 2] - a[j - 1];
                  else if (expression[i] == '*') a[j - 2] = a[j - 2] * a[j - 1];
                  else if (expression[i] == '/') a[j - 2] = a[j - 2] / a[j - 1];

                  j--;
            }
      }

      cout << a[0] << el;
}

3 - Write a program to convert an Infix expression into its equivalent Postfix expression.
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define el "\n"
#define sp " "

const int maxn = 1000000;
char a[maxn];
char s[maxn];
int top = -1;

int precedence(char c)
{
      if (c == '+' || c == '-') return 1;
      if (c == '*' || c == '/') return 2;
      return 0;
}

int main()
{
      string expression;
      getline(cin, expression);

      int j = 0;
      for (int i = 0; i < expression.length(); i++)
      {
            if (expression[i] == ' ') continue;
            else if (isalnum(expression[i]))
            {
                  a[j] = expression[i];
                  j++;
                  a[j] = ' ';
                  j++;
            }
            else if (expression[i] == '(')
            {
                  s[++top] = expression[i];
            }
            else if (expression[i] == ')')
            {
                  while (top >= 0 && s[top] != '(')
                  {
                        a[j] = s[top];
                        top--;
                        j++;
                        a[j] = ' ';
                        j++;
                  }
                  top--;
            }
            else
            {
                  while (top >= 0 && precedence(s[top]) >= precedence(expression[i]))
                  {
                        a[j] = s[top];
                        top--;
                        j++;
                        a[j] = ' ';
                        j++;
                  }
                  s[++top] = expression[i];
            }
      }

      while (top >= 0)
      {
            a[j] = s[top];
            top--;
            j++;
            a[j] = ' ';
            j++;
      }

      a[j] = '\0';
      cout << a << el;
}

4 - Write a program to calculate the Factorial of a number using recursive and non-recursive method
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define el "\n"
#define sp " "

long long factorial_recursive(int n)
{
      if (n == 0 || n == 1) return 1;
      return n * factorial_recursive(n - 1);
}

long long factorial_non_recursive(int n)
{
      long long result = 1;
      for (int i = 2; i <= n; i++)
            result *= i;
      return result;
}

int main()
{
      int n;
      cin >> n;

      cout << "Recursive Factorial: " << factorial_recursive(n) << el;
      cout << "Non-recursive Factorial: " << factorial_non_recursive(n) << el;
}

8 - Write a program to show the insert and delete operations of a circular queue.
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define el "\n"
#define sp " "

const int maxn = 5;
int queue[maxn], front = -1, rear = -1;

void enqueue(int value)
{
      if ((rear + 1) % maxn == front)
      {
            cout << "Queue is full." << el;
            return;
      }
      if (front == -1) front = 0;
      rear = (rear + 1) % maxn;
      queue[rear] = value;
}

void dequeue()
{
      if (front == -1)
      {
            cout << "Queue is empty." << el;
            return;
      }
      cout << "Dequeued element: " << queue[front] << el;
      if (front == rear) front = rear = -1;
      else front = (front + 1) % maxn;
}

void display()
{
      if (front == -1)
      {
            cout << "Queue is empty." << el;
            return;
      }
      int i = front;
      while (i != rear)
      {
            cout << queue[i] << sp;
            i = (i + 1) % maxn;
      }
      cout << queue[rear] << el;
}

int main()
{
      enqueue(10);
      enqueue(20);
      enqueue(30);
      enqueue(40);
      enqueue(50);
      enqueue(60); // Full queue

      display();
      dequeue();
      dequeue();

      display();
}

9 - Write a program to show the insert and delete operations of a priority queue using linked-list.
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define el "\n"
#define sp " "

struct Node
{
      int data;
      int priority;
      Node* next;
};

Node* front = nullptr;

void insert(int data, int priority)
{
      Node* temp = new Node();
      temp->data = data;
      temp->priority = priority;
      temp->next = nullptr;

      if (front == nullptr || front->priority < priority)
      {
            temp->next = front;
            front = temp;
      }
      else
      {
            Node* curr = front;
            while (curr->next != nullptr && curr->next->priority >= priority)
                  curr = curr->next;
            temp->next = curr->next;
            curr->next = temp;
      }
}

void deleteNode()
{
      if (front == nullptr)
      {
            cout << "Queue is empty." << el;
            return;
      }
      Node* temp = front;
      front = front->next;
      cout << "Deleted element: " << temp->data << el;
      delete temp;
}

void display()
{
      if (front == nullptr)
      {
            cout << "Queue is empty." << el;
            return;
      }
      Node* temp = front;
      while (temp != nullptr)
      {
            cout << temp->data << "(" << temp->priority << ")" << sp;
            temp = temp->next;
      }
      cout << el;
}

int main()
{
      insert(10, 2);
      insert(20, 1);
      insert(30, 3);
      display();
      deleteNode();
      display();
}

10 - Write a program to show the insert and delete operations of a priority queue using array.
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define el "\n"
#define sp " "

const int maxn = 5;
int queue[maxn], priorityQueue[maxn], size = 0;

void insert(int data, int priority)
{
      if (size == maxn)
      {
            cout << "Queue is full." << el;
            return;
      }

      int i = size - 1;
      while (i >= 0 && priorityQueue[i] < priority)
      {
            queue[i + 1] = queue[i];
            priorityQueue[i + 1] = priorityQueue[i];
            i--;
      }
      queue[i + 1] = data;
      priorityQueue[i + 1] = priority;
      size++;
}

void deleteNode()
{
      if (size == 0)
      {
            cout << "Queue is empty." << el;
            return;
      }

      cout << "Deleted element: " << queue[0] << el;
      for (int i = 0; i < size - 1; i++)
      {
            queue[i] = queue[i + 1];
            priorityQueue[i] = priorityQueue[i + 1];
      }
      size--;
}

void display()
{
      if (size == 0)
      {
            cout << "Queue is empty." << el;
            return;
      }

      for (int i = 0; i < size; i++)
      {
            cout << queue[i] << "(" << priorityQueue[i] << ")" << sp;
      }
      cout << el;
}

int main()
{
      insert(10, 2);
      insert(20, 1);
      insert(30, 3);
      display();
      deleteNode();
      display();
}

11 - Write a program that will read a positive integer in base b (2 <= b <= 16) and convert it into base d (2 <= d <= 16 ).
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define el "\n"
#define sp " "

int convert_to_decimal(string number, int base)
{
      int result = 0;
      int power = 1;
      for (int i = number.size() - 1; i >= 0; i--)
      {
            if (number[i] >= '0' && number[i] <= '9')
                  result += (number[i] - '0') * power;
            else if (number[i] >= 'A' && number[i] <= 'F')
                  result += (number[i] - 'A' + 10) * power;
            power *= base;
      }
      return result;
}

void convert_from_decimal(int num, int base)
{
      string result = "";
      while (num > 0)
      {
            int remainder = num % base;
            if (remainder < 10)
                  result += char(remainder + '0');
            else
                  result += char(remainder - 10 + 'A');
            num /= base;
      }
      reverse(result.begin(), result.end());
      cout << result << el;
}

int main()
{
      string number;
      int base_b, base_d;
      cin >> number >> base_b >> base_d;

      int decimal_value = convert_to_decimal(number, base_b);
      convert_from_decimal(decimal_value, base_d);
}

12 - Write a program to determine the Greatest Common Divisor (GCD) & Least Common Multiple (LCM) of two given positive integers.
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define el "\n"
#define sp " "

int gcd(int a, int b)
{
      if (b == 0) return a;
      return gcd(b, a % b);
}

int lcm(int a, int b)
{
      return (a * b) / gcd(a, b);
}

int main()
{
      int a, b;
      cin >> a >> b;

      cout << "GCD: " << gcd(a, b) << el;
      cout << "LCM: " << lcm(a, b) << el;
}
