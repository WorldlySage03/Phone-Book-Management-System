1. Write a program to sort n numbers using Insertion Sort algorithm.
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define sp " "
#define el '\n'

int main()
{
      int n;
      cin >> n;

      int a[n];
      for (int i = 0; i < n; i++)
      {
            cin >> a[i];
      }

      for (int i = 1; i < n; i++)
      {
            int j = i - 1, k = a[i];
            while (j >= 0 && a[j] >= k)
            {
                  a[j + 1] = a[j];
                  j--;
            }
            a[j + 1] = k;
      }

      for (int i = 0; i < n; i++)
      {
            cout << a[i] << sp;
      }
      cout << el;

      return 0;
}

            int m = a[i], p = i;

#include <bits/stdc++.h>
using namespace std;

#define sp " "
#define el '\n'

int main()
{
    int n1, n2;
    cin >> n1;
    int a[n1];
    for (int i = 0; i < n1; i++)
        cin >> a[i];

    cin >> n2;
    int b[n2];
    for (int i = 0; i < n2; i++)
        cin >> b[i];

    int result[n1 + n2];
    int i = 0, j = 0, k = 0;

    while (i < n1 && j < n2)
    {
        if (a[i] <= b[j])
            result[k++] = a[i++];
        else
            result[k++] = b[j++];
    }

    while (i < n1)
        result[k++] = a[i++];

    while (j < n2)
        result[k++] = b[j++];

    for (int i = 0; i < n1 + n2; i++)
    {
        cout << result[i] << sp;
    }
    cout << el;

    return 0;
}

#include <bits/stdc++.h>
using namespace std;

#define sp " "
#define el '\n'

void merge(int a[], int l, int mid, int r)
{
      int n1 = mid - l + 1;
      int n2 = r - mid;

      int t1[n1], t2[n2];

      for (int i = 0; i < n1; i++)
            t1[i] = a[i + l];
      for (int i = 0; i < n2; i++)
            t2[i] = a[i + mid + 1];

      int i = 0, j = 0, k = l;

      while (i < n1 && j < n2)
      {
            if (t1[i] <= t2[j]) a[k++] = t1[i++];
            else a[k++] = t2[j++];
      }
      while (i < n1) a[k++] = t1[i++];
      while (j < n2) a[k++] = t2[j++];
}

void mergeSort(int a[], int l, int r)
{
      if (l < r)
      {
            int mid = l + (r - l) / 2;

            mergeSort(a, l, mid);
            mergeSort(a, mid + 1, r);

            merge(a, l, mid, r);
      }
}

int main()
{
      int n;
      cin >> n;

      int a[n];
      for (int i = 0; i < n; i++)
      {
            cin >> a[i];
      }

      mergeSort(a, 0, n - 1);

      for (int i = 0; i < n; i++)
      {
            cout << a[i] << sp;
      }
      cout << el;

      return 0;
}

      if (root == nullptr) return false;

            int t;
            cin >> t;

            root = insert(root, t);

      display(root);
      cout << el;

      int ele;
      cin >> ele;

      if (search(root, ele)) cout << "Element " << ele << " exists." << el;
      else cout << "Element " << ele << " does not exist." << el;

      newNode->num = x;
      newNode->left = newNode->right = nullptr;
      return newNode;
}

BstNode* insert(BstNode* rootptr, int x)
{
      if (rootptr == nullptr) rootptr = createNode(x);
      else if (x <= rootptr->num) rootptr->left = insert(rootptr->left, x);
      else  rootptr->right = insert(rootptr->right, x);
      return rootptr;

void display(BstNode* rootptr)
{
      if (rootptr != nullptr)
      {
            display(rootptr->left);
            cout << rootptr->num << sp;
            display(rootptr->right);
      }
}

BstNode* FindMin(BstNode* rootptr)
{
      while (rootptr->left != nullptr) rootptr = rootptr->left;
      return rootptr;
}

BstNode* del(BstNode* rootptr, int x)
{
      if (rootptr == nullptr) return rootptr;
      else if (x < rootptr->num) rootptr->left = del(rootptr->left, x);
      else if (x > rootptr->num) rootptr->right = del(rootptr->right, x);
      else
      {
            if (rootptr->left == nullptr && rootptr->right == nullptr)
            {
                  delete rootptr;
                  rootptr = nullptr;
            }
            else if (rootptr->left == nullptr)
            {
                  BstNode* t = rootptr;
                  rootptr = rootptr->right;
                  delete t;
            }
            else if (rootptr->right == nullptr)
            {
                  BstNode* t = rootptr;
                  rootptr = rootptr->left;
                  delete t;
            }
            else
            {
                  BstNode* t = FindMin(rootptr->right);
                  rootptr->num = t->num;
                  rootptr->right = del(rootptr->right, t->num);
            }
      }
      return rootptr;
}

int main()
{
      BstNode* rootptr = nullptr;

      int n;
      cin >> n;

      for (int i = 0; i < n; i++)
      {
            int x;
            cin >> x;

            rootptr = insert(rootptr, x);
      }

      display(rootptr);
      cout << el;

      int x;
      cout << "Element to delete: ";
      cin >> x;

      rootptr = del(rootptr, x);

      display(rootptr);
}

11. Write a program to sort n numbers using Heap sort algorithm.
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define sp " "
#define el '\n'

void heapify(int a[], int i)
{
      for (; i > 0; i = (i - 1) / 2)
      {
            if (a[(i - 1) / 2] < a[i])
                  swap(a[(i - 1) / 2], a[i]);
            else break;
      }
}

void heapSort(int a[], int n)
{
      while (n--)
      {
            swap(a[0], a[n]);
            for (int i = 0; i < n;)
            {
                  if (i * 2 + 2 < n && a[i * 2 + 1] >= a[i * 2 + 2] && a[i] < a[i * 2 + 1])
                  {
                        swap(a[i], a[i * 2 + 1]);
                        i = i * 2 + 1;
                  }
                  else if (i * 2 + 2 < n && a[i * 2 + 1] < a[i * 2 + 2] && a[i] < a[i * 2 + 2])
                  {
                        swap(a[i], a[i * 2 + 2]);
                        i = i * 2 + 2;
                  }
                  else if (i * 2 + 1 < n && a[i] < a[i * 2 + 1])
                  {
                        swap(a[i], a[i * 2 + 1]);
                        i = i * 2 + 1;
                  }
                  else break;
            }
      }
}

int main()
{
      int n;
      cin >> n;

      int a[n];
      for (int i = 0; i < n; i++)
      {
            cin >> a[i];
            heapify(a, i);
      }

      for (int i = 0, k = 1, t = 0; i < n; k *= 2, t++)
      {
            cout << "Step - " << t << ": ";
            for (int j = 0; i < n && j < k; i++, j++)
            {
                  cout << a[i] << sp;
            }
            cout << el;
      }
      cout << el;

      heapSort(a, n);

      cout << "Sorted Array: ";
      for (int i = 0; i < n; i++)
      {
            cout << a[i] << sp;
      }
      cout << el;
}

#include <bits/stdc++.h>
using namespace std;

#define sp " "
#define el '\n'

void MatrixMlt(int a[100][100], int aa[100][100], int n)
{
      int result[100][100];

      for (int i = 0; i < n; i++)
      {
            for (int j = 0; j < n; j++)
            {
                  int t = 0;
                  for (int k = 0; k < n; k++)
                  {
                        t += a[i][k] * aa[k][j];
                  }
                  result[i][j] = t;
            }
      }

      for (int i = 0; i < n; i++)
      {
            for (int j = 0; j < n; j++)
            {
                  aa[i][j] = result[i][j];
            }
      }
}

int main()
{
      int n, m;

      cout << "Input the number of nodes: ";
      cin >> n;
      cout << "Input the number of edges: ";
      cin >> m;

      int a[100][100], aa[100][100], sum[100][100];
      for (int i = 0; i < n; i++)
      {
            for (int j = 0; j < n; j++)
            {
                  a[i][j] = aa[i][j] = sum[i][j] = 0;
            }
      }

      cout << "Input the edges: " << el;
      for (int i = 0; i < m; i++)
      {
            int t1, t2;
            cin >> t1 >> t2;

            a[t1 - 1][t2 - 1] = aa[t1 - 1][t2 - 1] = sum[t1 - 1][t2 - 1] = 1;
      }

      cout << el << "Adjacency Matrix:" << el;
      for (int i = 0; i < n; i++)
      {
            for (int j = 0; j < n; j++)
            {
                  cout << a[i][j] << sp;
            }
            cout << el;
      }

      for (int i = 0; i < n - 1; i++)
      {
            MatrixMlt(a, aa, n);

            for (int j = 0; j < n; j++)
            {
                  for (int k = 0; k < n; k++)
                  {
                        sum[j][k] += aa[j][k];
                  }
            }
      }

      cout << el << "Path Matrix:" << el;
      for (int i = 0; i < n; i++)
      {
            for (int j = 0; j < n; j++)
            {
                  if (sum[i][j] != 0) cout << 1 << sp;
                  else cout << 0 << sp;
            }
            cout << el;
      }
}

#include <bits/stdc++.h>
using namespace std;

#define sp " "
#define el '\n'

void Display(int a[100][100], int n)
{
      for (int i = 0; i < n; i++)
      {
            for (int j = 0; j < n; j++)
            {
                  cout << a[i][j] << sp;

            cout << el;

void Warshall(int a[100][100], int n)
{
      for (int k = 0; k < n; k++)
      {
            for (int i = 0; i < n; i++)
            {
                  for (int j = 0; j < n; j++)
                  {
                        a[i][j] = a[i][j] | (a[i][k] & a[k][j]);
                  }
            }
      }
}

int main()
{
      int n, m;

      cout << "Input the number of nodes: ";
      cin >> n;
      cout << "Input the number of edges: ";
      cin >> m;

      int a[100][100];
      for (int i = 0; i < n; i++)
      {
            for (int j = 0; j < n; j++)
            {
                  a[i][j] = 0;
            }
      }

      cout << "Input the edges: " << el;
      for (int i = 0; i < m; i++)
      {
            int t1, t2;
            cin >> t1 >> t2;

            a[t1 - 1][t2 - 1] = 1;
      }

      cout << el << "Adjacency Matrix:" << el;
      Display(a, n);

      Warshall(a, n);

      cout << el << "Path Matrix:" << el;
      Display(a, n);
}

16. Write a program to traverse a graph using Breadth First Search.
--------------------------------------------------

#include <bits/stdc++.h>
using namespace std;

#define sp " "
#define el '\n'

void BFS(vector<int> adj[], int start, int n)
{
      vector<bool> visited(n);
      queue<int> q;

      visited[start] = 1;
      q.push(start);

      cout << el << "BFS Traversal:" << el;

      while (!q.empty())
      {
            int u = q.front();
            q.pop();

            cout << u + 1 << sp;

            for (int v: adj[u])
            {
                  if (!visited[v])
                  {
                        visited[v] = true;
                        q.push(v);
                  }
            }
      }
      cout << el;
}

int main()
{
      int n, m;

      cout << "Input the number of nodes: ";
      cin >> n;
      cout << "Input the number of edges: ";
      cin >> m;

      vector<int> adj[n];
      for (int i = 0; i < m; i++)
      {
            int u, v;
            cin >> u >> v;

            adj[u - 1].push_back(v - 1);
            adj[v - 1].push_back(u - 1);
      }

      int start;
      cout << "Input the starting node for BFS: ";
      cin >> start;

      BFS(adj, start - 1, n);
}

#include <bits/stdc++.h>
using namespace std;

#define sp " "
#define el '\n'

void DFS(vector<int> adj[], int start, int n)
{
      vector<bool> visited(n);
      stack<int> s;

      s.push(start);

      cout << el << "DFS Traversal:" << el;

      while (!s.empty())
      {
            int u = s.top();
            s.pop();

            if (!visited[u])
            {
                  cout << u + 1 << sp;
                  visited[u] = true;
            }

            for (int v: adj[u])
            {
                  if (!visited[v])
                        s.push(v);
            }
      }
      cout << el;
}

int main()

{

            a[i] = -1;

#include <bits/stdc++.h>
using namespace std;

#define sp " "
#define el '\n'

struct Node
{
      int num;
      struct Node *next;
};

int main()
{
      int n;
      cin >> n;

      Node *a[n] = {};

      for (int i = 0; i < n; i++)
      {
            int t;
            cin >> t;
            
            int index = t % n;
            if (a[index] == nullptr)
            {
                  a[index] = new Node();
                  a[index]->num = t;
                  a[index]->next = nullptr;
            }
            else
            {
                  Node *ptr = a[index];
                  while (ptr->next != nullptr)
                        ptr = ptr->next;

                  ptr->next = new Node();
                  ptr = ptr->next;
                  ptr->num = t;
                  ptr->next = nullptr;
            }
      }
      
      for (int i = 0; i < n; i++)
      {
            cout << "Index " << i << ": ";
            
            Node* ptr = a[i];
            while (ptr != nullptr)
            {
                  cout << ptr->num << " -> ";
                  ptr = ptr->next;
            }
            cout << "NULL" << el;
      }
}
